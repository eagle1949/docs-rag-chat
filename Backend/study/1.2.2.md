## 8. 流式处理 (Streaming)

### 8.1 LLM Streaming

```python
# 同步流式
for chunk in llm.stream("你好"):
    print(chunk.content, end="", flush=True)

# 异步流式
async for chunk in llm.astream("你好"):
    print(chunk.content, end="", flush=True)
```

### 8.2 Chain Streaming

```python
# 基础流式
for chunk in chain.stream({"input": "hello"}):
    print(chunk, end="", flush=True)

# 事件流 (更细粒度)
for event in chain.astream_events({"input": "hello"}, version="v1"):
    if event["event"] == "on_llm_stream":
        print(event["data"]["chunk"].content)
```

## 18. 版本迁移指南

### 18.1 从 0.x 到 1.x 的重大变化

#### Memory系统 🔥

```python
# ❌ 0.x 旧方式
from langchain.memory import ConversationBufferMemory
memory = ConversationBufferMemory()
chain = ConversationChain(llm=llm, memory=memory)

# ✅ 1.x 新方式
from langchain_core.chat_history import InMemoryChatMessageHistory
from langchain_core.runnables.history import RunnableWithMessageHistory

store = {}
def get_session_history(session_id):
    if session_id not in store:
        store[session_id] = InMemoryChatMessageHistory()
    return store[session_id]

chain_with_history = RunnableWithMessageHistory(
    chain,
    get_session_history,
    input_messages_key="input",
    history_messages_key="history"
)
```

#### Chain构建 🔥

```python
# ❌ 0.x 旧方式
from langchain.chains import LLMChain
chain = LLMChain(llm=llm, prompt=prompt)

# ✅ 1.x 新方式 (LCEL)
chain = prompt | llm | StrOutputParser()
```

#### Agent创建 🔥

```python
# ❌ 0.x 旧方式
from langchain.agents import initialize_agent, AgentType
agent = initialize_agent(
    tools,
    llm,
    agent=AgentType.ZERO_SHOT_REACT_DESCRIPTION
)

# ✅ 1.x 新方式
from langchain.agents import create_tool_calling_agent, AgentExecutor
agent = create_tool_calling_agent(llm, tools, prompt)
agent_executor = AgentExecutor(agent=agent, tools=tools)
```

#### 工具加载 🔥

```python
# ❌ 0.x 旧方式
from langchain.agents import load_tools
tools = load_tools(["wikipedia", "llm-math"], llm=llm)

# ✅ 1.x 新方式
from langchain_community.tools import WikipediaQueryRun
from langchain_community.utilities import WikipediaAPIWrapper

wikipedia = WikipediaQueryRun(api_wrapper=WikipediaAPIWrapper())
tools = [wikipedia, ...]
```

from langchain_core.prompts import ChatPromptTemplate, MessagesPlaceholder

### 19.3 RAG应用

```python
from langchain_community.document_loaders import TextLoader
from langchain_text_splitters import RecursiveCharacterTextSplitter
from langchain_openai import OpenAIEmbeddings
from langchain_community.vectorstores import Chroma
from langchain_core.runnables import RunnablePassthrough

# 加载文档
loader = TextLoader("data.txt")
documents = loader.load()

# 分割
text_splitter = RecursiveCharacterTextSplitter(chunk_size=1000)
splits = text_splitter.split_documents(documents)

# 向量化
embeddings = OpenAIEmbeddings()
vectorstore = Chroma.from_documents(splits, embeddings)

# 检索器
retriever = vectorstore.as_retriever()

# RAG链
prompt = ChatPromptTemplate.from_template("""
基于以下上下文回答问题:
{context}

问题: {question}
""")

chain = (
    {"context": retriever, "question": RunnablePassthrough()}
    | prompt
    | llm
    | StrOutputParser()
)

result = chain.invoke("问题是什么?")
```

### 19.4 Agent应用

```python
from langchain.agents import create_tool_calling_agent, AgentExecutor
from langchain_community.tools import DuckDuckGoSearchRun

# 创建工具
search = DuckDuckGoSearchRun()
tools = [search]

# 创建prompt
prompt = ChatPromptTemplate.from_messages([
    ("system", "你是一个助手，可以使用工具。"),
    ("human", "{input}"),
    MessagesPlaceholder(variable_name="agent_scratchpad"),
])

# 创建agent
agent = create_tool_calling_agent(llm, tools, prompt)
agent_executor = AgentExecutor(agent=agent, tools=tools, verbose=True)

# 执行
result = agent_executor.invoke({"input": "搜索最新AI新闻"})
```

### 19.5 多链条件路由

```python
from langchain_core.runnables import RunnableBranch

# 分类链
classification_chain = (
    ChatPromptTemplate.from_template("分类问题: {question}\n类别: ")
    | llm
    | StrOutputParser()
)

# 不同类型的处理链
general_chain = ChatPromptTemplate.from_template("一般回答: {question}") | llm
tech_chain = ChatPromptTemplate.from_template("技术回答: {question}") | llm

# 路由
branch = RunnableBranch(
    (lambda x: "技术" in x["type"], tech_chain),
    general_chain  # 默认
)

# 组合
full_chain = (
    {"question": RunnablePassthrough(), "type": classification_chain}
    | branch
)

result = full_chain.invoke("什么是Python?")
```
